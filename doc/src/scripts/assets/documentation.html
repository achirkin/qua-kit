RemoteDeregister

Deregisters a client from a service registration. This will cancel any running service call, cancel all subscriptions of this client and, if the service to be deregistered is the last remaining of its kind, its serviceName will be removedfrom the list of available services.

inputs:

  • run: "RemoteDeregister"

outputs:

  • XOR error: "string",
  • XOR result: {
    • deregisteredName: "string", // Intended to inform listeners about the service name that was deregistered
    • id: "number", // int, the id that was deregistered
    • nodeIP: "string", // Intended to inform listeners about the IP of the machine that deregistered the service
    • remainsAvailable: "boolean" // boolean to indicate whether other services with the samename remain available (true) or whether the service is being deleted from the listof available services (false)
    }

example call:

    {
  • run: "RemoteDeregister"
    • }

RemoteRegister

Registers a client as a service.

inputs:

  • run: "RemoteRegister",
  • description: "string",
  • exampleCall: "json", // Mandatory json object that shows an example call. Since OPT/XOR/ANY modifiers are only allowed in in & output specifications and not in calls, an example call shows very clearly how a call could look like.
  • serviceName: "string", // mandatory string that identifies the service
  • OPT id: "number", // Optional int to be used to identify a remote service. E.g. for viewer services it might be important to be able to identify remote service instances. If the id is taken already an error will be thrown / registration aborted.
  • OPT inputs: "json", // Optional json object holding the input description to be included by 'ServiceInfo' / the API
  • OPT outputs: "json" // Optional json object holding the output description to be included by 'ServiceInfo' / the API

outputs:

  • XOR error: "string",
  • XOR result: {
    • id: "number", // The int given as input or an int generated by Luci
    • nodeIP: "string", // The IP of the client that was registered as a service (intended to inform listeners)
    • registeredName: "string" // Intended to inform listeners about the service name that was registered
    }

example call:

    {
  • run: "RemoteRegister",
  • exampleCall: {
    • run: "example",
    • inputs: {
      • namedInput1: 4
      }
    },
  • inputs: {
    • namedInput1: "number"
    },
  • serviceName: "example"
    • }

ServiceControl

Used to remotely administrate installed services on a selected machine or on allmachines having a 'ServiceControl' remote service running.

inputs:

  • run: "ServiceControl",
  • XOR info: [ // get filtered information about either selected machines (nodes) or about selectedserviceNames on all machines or full information about all machines in case this value is 'null'
    • {
      • XOR nodes: ["string"],
      • XOR serviceNames: ["string"],
      • OPT threadPool: "boolean"
      },
    • "null"
    ],
  • XOR install: { // install files on either selected machines (nodes) or on all machines (services)
    • XOR nodes: { // values for 'ANY IP' or 'services' is a list holding only attachments (and no other type)
      • ANY IP: ["attachment"] // the attachment must hold attributes: {'exec': 'string (the string to start the service in console)','numRequestedCPUCores':'number (0=all available cores)'}
      },
    • XOR services: ["attachment"]
    },
  • XOR remove: { // remove services identified by their name (serviceName) either on selected machines (nodes) or on all machines (services)
    • XOR nodes: {
      • ANY IP: ["string"]
      },
    • XOR services: ["string"]
    },
  • XOR start: { // startup a given number of services either globally or on selected machines (nodes)
    • XOR nodes: {
      • ANY IP: {
        • ANY serviceName: [ // the service to start on a machine: either pass a number to specify how many services should be started or pass a json object consisting of id (=instanceID or persistentID) and the corresponding args stringto be passed on startup
          • "number",
          • {
            • ANY id: "string"
            }
          ]
        }
      },
    • XOR services: {
      • ANY serviceName: "number"
      }
    },
  • XOR stop: { // stop a given amount of services either globally or on specified machines (nodes).
    • XOR nodes: {
      • ANY IP: {
        • ANY serviceName: "number"
        }
      },
    • XOR services: {
      • ANY serviceName: "number"
      }
    }

outputs:

  • XOR error: "string",
  • XOR result: {
    • XOR info: {
      • XOR nodes: {
        • ANY ip: {
          • services: {
            • busy: {
              • ANY serviceName: "number"
              },
            • idle: {
              • ANY serviceName: "number"
              },
            • running: {
              • ANY serviceName: "number"
              }
            },
          • OPT installedServices: {
            • ANY serviceName: {
              • exec: "string", // to be run in console
              • numRequestedCPUCores: "number" // 0 = using all available cores
              }
            },
          • OPT system: {
            • CPU: {
              • archNumBits: "number", // int; 32 or 64
              • endian: "string", // 'big' or 'little'
              • numCores: "number"
              },
            • name: "string",
            • osname: "string",
            • osversion: "string"
            }
          }
        },
      • XOR services: {
        • ANY serviceName: {
          • avgDuration: "number", // in milliseconds
          • numCalls: "number", // how many times this service has been called already
          • OPT available: "number", // not busy
          • OPT nodes: "list", // list of IPv4
          • OPT running: "number"
          }
        },
      • remoteSummary: {
        • busy: "number",
        • idle: "number"
        },
      • OPT quickPool: { // thread pool that handles all services (threads) having an avgDuration < 2 seconds
        • avgWaitingTime: "number",
        • maxPoolSize: "number",
        • numCalls: "number",
        • poolSize: "number",
        • waitingInQueue: "number"
        },
      • OPT slowPool: { // thread pool that handles all services (threads) having an avgDuration > 2 seconds
        • avgWaitingTime: "number",
        • maxPoolSize: "number",
        • numCalls: "number",
        • poolSize: "number",
        • waitingInQueue: "number"
        }
      },
    • XOR installed: {
      • ANY IP: ["string"] // list of service names installed per IP
      },
    • XOR removed: {
      • ANY IP: ["string"] // list of services names removed per IP
      },
    • XOR started: {
      • ANY IP: { // number of services started per IP and service
        • ANY serviceName: "number"
        }
      },
    • XOR stopped: {
      • ANY IP: { // number of services stopped per IP and service
        • ANY serviceName: "number"
        }
      },
    • OPT errors: "list" // 'TODO' in case at least one of the items to install/remove/start/stop produced an error this list contains one error string per item
    }

example call:

    {
  • run: "ServiceControl",
  • info: ,
  • serviceName: "ServiceControl"
    • }

task.Get

Returns infos about a task (requesting a taskID) or returns a list if ids representingthe same service

inputs:

  • run: "task.Get",
  • XOR serviceName: "string", // requests a list of taskID representing the same service
  • XOR taskID: "number" // requests infos about a specific task

outputs:

  • XOR error: "string",
  • XOR result: {
    • XOR task: {
      • inputSchema: "json", // the current input values
      • inputs: "json", // the inputs with subscriptions being represented as {'taskID':'number','key':'string (the name of the output key of the task referenced by taskID)'}
      • listensToDone: "list", // a list of taskIDs the task is listening to generally (=not to specific outputs)
      • name: "string", // can be different from service name
      • parentID: "number", // the taskID of a workflow to which this task belongs; 0 = no parent ID / root workflow
      • position: "json", // {'x':'number','y':'number'}
      • taskID: "number",
      • OPT elements: "list", // if the task is a workflow this list contains all children
      • OPT services: "list" // if the task is a workflow this list contains all services that need to started up (if any are set)
      },
    • XOR taskIDs: "list"
    }

example call:

    {
  • run: "task.Get",
  • taskID: 3
    • }
